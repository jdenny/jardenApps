Booking
    attributes: personCount, startDate
    methods: reserve(), cancel(), confirm(), revise(), checkOut()

HotelBooking extends Booking
    attributes: endDate, roomType
    methods: checkOut() - take payment

SportsBooking extends Booking
    activityType (pool, tennis, sauna, massage, gym), time
    methods: checkOut() - some activities may be free;
        for those charged, put on room bill

RestaurantBooking extends Booking
    attributes: time
    methods: checkOut() - could pay now or put on room bill
